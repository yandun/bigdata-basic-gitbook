zk02 

wget http://mirror.bit.edu.cn/apache/zookeeper/zookeeper-3.4.14/zookeeper-3.4.14.tar.gz

hostnamectl  set-hostname zk02

sed -i 's/enforcing/disabled/g'  /etc/selinux/config
setenforce 0

sed -i 's/#UseDNS yes/UseDNS no/g'   /etc/ssh/sshd_config
systemctl   restart sshd

systemctl  disable firewalld NetworkManager
systemctl  stop  firewalld  NetworkManager

yum -y install ntpdate
ntpdate ntp1.aliyun.com

yum -y install ntp
systemctl  enable  ntpd

cat >   /etc/ntp.conf  << EOF
driftfile /var/lib/ntp/drift
restrict default nomodify notrap nopeer noquery
restrict 127.0.0.1 
restrict ::1
server ntp1.aliyun.com iburst  iburst
logfile /var/log/ntp.log
includefile /etc/ntp/crypto/pw
keys /etc/ntp/keys
disable monitor
EOF

systemctl  start   ntpd
ntpq -p

cat > /etc/resolv.conf << EOF
nameserver 223.5.5.5
EOF

cat /etc/hosts
127.0.0.1   localhost localhost.localdomain localhost4 localhost4.localdomain4
::1         localhost localhost.localdomain localhost6 localhost6.localdomain6
172.16.99.101  zk01
172.16.99.102  zk02
172.16.99.103  zk03

tar xzvf /opt/zookeeper-3.4.14.tar.gz   -C /usr/local/

/etc/profile

export ZOOKEEPER_HOME=/usr/local/zookeeper-3.4.14
export PATH=$PATH:$ZOOKEEPER_HOME/bin

source  /etc/profile

mkdir -p /zookeeper/data/
mkdir -p /zookeeper/logs

echo 2  > /zookeeper/data/myid

cat > /usr/local/zookeeper-3.4.14/conf/zoo.cfg << EOF
tickTime=2000
initLimit=10
syncLimit=5
clientPort=2181
dataDir=/zookeeper/data
dataLogDir=/zookeeper/logs
server.1=zk01:2888:3888
server.2=zk02:2888:3888
server.3=zk03:2888:3888
EOF

zkServer.sh start

zkServer.sh  status

mkdir -p /opt/hadoop/journal
